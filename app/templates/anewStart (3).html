def cutWords(msg, stopWords):
    seq = jieba.cut(msg)
    leftWords = ""
    for item in seq:
        if item not in stopWords:
            leftWords += item + " "
    return leftWords

    for sentence in dataFrame['comment-cut']:
        sentence = sentence.split()
        for item in sentence:
            if len(item) == 1:
                continue
            if item in worddict.keys():
                worddict[item] += 1
            else:
                worddict[item] = 1
#     with open(outputfile, "w") as file:
